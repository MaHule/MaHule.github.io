利用回文字符串翻转仍然是该字符串可以判断是否是回文字符串

9.将列表展开

#from iteration_utilities import deepflatten

#方法1
list17 = [[12,5,3],[2,4,[5],[6,9,7]],[5,8,[9,[10,12]]]]

#方法2
def flatten(lst):
    res = []
    for i in lst:
        if isinstance(i,list):
            res.extend(flatten(i))
        else:
            res.append(i)
    return res

print(flatten(list17))

list10 = ['p','p','Y','y','t','t','h','o','o','o','n']

print(count['p'])

    if i in dict1:
